❯ 项目目录/home/lg/0EGG # TORK 裸机核心架构规范 v0.1                            
                                                                                
  ## 任务                                                                       
  编写 tork_core.asm，一个 x86-64 Linux 用户态汇编程序。                        
  这不是一个 shell 工具，不是一个编译器，甚至不是传统意义上的"程序"。           
  它是一颗心脏——唯一的工作是保持稳定的节律，并保护一个 96 字节的数据块。        
                                                                                
  ## 行为规范                                                                   
                                                                                
  ### 1. 心跳循环                                                               
  - 程序入口 _start 后立即进入主循环                                            
  - 每圈循环称为一个 "tick"                                                     
  - 每个 tick 的执行顺序：感知 → 比较 → 反应 → 记录 → 等待 → 下一圈             
  - 目标心跳间隔：100ms（即 10 tick/秒）                                        
  - 使用 RDTSC 指令测量时间，不依赖 usleep 或任何系统调用                       
  - 初始 TSC 频率假设 3.0GHz（后续会校准，但第一版用常数）                      
                                                                                
  ### 2. 灵魂（Soul）结构体                                                     
  - 映射到固定地址 0x200000（使用 mmap，MAP_FIXED | MAP_ANONYMOUS |             
  MAP_PRIVATE）                                                                 
  - 大小：96 字节                                                               
  - 布局如下（偏移量从 0 开始）：                                               
                                                                                
  偏移 0x00 (4字节): tick        — uint32, 当前 tick 数，每循环 +1              
  偏移 0x04 (8字节): last_tsc    — uint64, 上一个 tick 开始时的 TSC 值          
  偏移 0x0C (8字节): current_tsc — uint64, 当前 tick 的 TSC 值                  
  偏移 0x14 (8字节): elapsed_tsc — uint64, 当前 tick 与上一个 tick 的 TSC 差值  
  偏移 0x1C (8字节): expected_tsc— uint64, 期望的心跳间隔（TSC 计数）           
  偏移 0x24 (1字节): hw_stress   — uint8,  硬件压力 0-3，初始 0                 
  偏移 0x25 (1字节): mode        — uint8,  模态 0=BUSY/1=IDLE/2=SLEEP，初始 0   
  偏移 0x26 (2字节): reserved    — 填充                                         
  偏移 0x28 (4字节): checksum    — uint32, 对整个 96 字节的 CRC32（自身         
  checksum 字段置零后计算）                                                     
                                                                                
  剩余 64 字节全部保留为 0x00，预留给未来扩展。                                 
  总计 96 字节。                                                                
                                                                                
  ### 3. Checksum 保护                                                          
  - 每次写 soul 后必须更新 checksum                                             
  - 每次读 soul 前必须验证 checksum                                             
  - 不匹配时：将 soul 所有字节清零，tick 从 0 重新开始                          
  - CRC32 算法使用 x86 CRC32 指令（SSE                                          
  4.2）或软件实现（软件实现即可，第一版简单优先）                               
                                                                                
  ### 4. 感知（Sense）                                                          
  每个 tick 开始时：                                                            
    1. tick++                                                                   
    2. 读取当前 TSC（rdtscp 序列化指令）                                        
    3. 计算 elapsed_tsc = current_tsc - last_tsc                                
    4. 将 current_tsc 保存到 soul                                               
                                                                                
  ### 5. 比较（Compare）                                                        
    1. 计算偏差 = elapsed_tsc - expected_tsc                                    
    2. 如果偏差绝对值 > expected_tsc * 0.1（即 > 10% 偏差）:                    
       将 hw_stress 设为 1（轻度异常）                                          
                                                                                
  ### 6. 反应（React）                                                          
    1. 验证 soul checksum                                                       
    2. 如果 checksum 不匹配：清零 soul，tick 从 0 开始                          
    3. 更新 last_tsc = current_tsc                                              
                                                                                
  ### 7. 记录（Record）                                                         
    1. 更新 soul checksum                                                       
                                                                                
  ### 8. 等待（Wait）                                                           
    1. 计算需要等待的 TSC 计数：                                                
       wait_tsc = expected_tsc - (当前 TSC - last_tsc)                          
    2. 如果 wait_tsc > 0：执行 busy-wait 循环（使用 pause 指令降低功耗）        
    3. 不要使用系统调用 sleep——这个循环不能交出控制权                           
                                                                                
  ### 9. 输出（第一版调试用）                                                   
  每个 tick 向 stdout 写入一行：".", 每 100 个 tick 写入一行 "tick N"           
  使用 write 系统调用（rax=1, rdi=1, rsi=buf, rdx=len）                         
                                                                                
  ## 编译方式                                                                   
    as -o tork_core.o tork_core.asm                                             
    ld -o tork_core tork_core.o                                                 
                                                                                
  ## 验收标准                                                                   
  1. 运行 ./tork_core，5 秒内输出约 50 个 "." 和 "tick 100" 字样                
  2. kill -9 后重新运行，tick 从 0 开始（重启不保留状态）                       
  3. 检查 /proc/PID/maps 中 0x200000 处有 4KB rw 映射                           
  4. 用 xxd /proc/PID/mem 读取偏移 0x200000 处验证 soul 结构                    
  5. 长时间运行（30 秒）无崩溃

  tork_core.asm 源代码

  # TORK Core - Bare Metal Heartbeat
  # x86-64 Linux userspace, no libc
  # Assemble: as -o tork_core.o tork_core.asm
  # Link:     ld -o tork_core tork_core.o

  # ── Constants ──────────────────────────────────────────────────────
  .equ SOUL_ADDR,      0x200000
  .equ SOUL_SIZE,      96
  .equ PAGE_SIZE,      4096
  .equ EXPECTED_TSC,   300000000        # 3 GHz × 100 ms

  # Soul layout
  .equ S_TICK,         0x00             # uint32
  .equ S_LAST_TSC,     0x04             # uint64
  .equ S_CUR_TSC,      0x0C             # uint64
  .equ S_ELAPSED,      0x14             # uint64
  .equ S_EXPECTED,     0x1C             # uint64
  .equ S_HW_STRESS,    0x24             # uint8
  .equ S_MODE,         0x25             # uint8
  .equ S_PAD,          0x26             # 2 bytes
  .equ S_CRC,          0x28             # uint32

  # syscalls
  .equ SYS_WRITE,      1
  .equ SYS_MMAP,       9
  .equ SYS_EXIT,       60

  # mmap flags
  .equ PROT_RW,        0x03
  .equ MAP_FPA,        0x32             # MAP_FIXED|MAP_PRIVATE|MAP_ANONYMOUS

  .section .rodata
  dot:       .byte '.'
  prefix:    .ascii "tick "

  .section .bss
     .align 8
  tbuf:      .space 32

  .section .text
  .globl _start

  # ══════════════════════════════════════════════════════════════════
  # Entry
  # ══════════════════════════════════════════════════════════════════
  _start:
      # mmap soul at 0x200000
      movq $SYS_MMAP, %rax
      movq $SOUL_ADDR, %rdi
      movq $PAGE_SIZE, %rsi
      movq $PROT_RW, %rdx
      movq $MAP_FPA, %r10
      movq $-1, %r8
      xorq %r9, %r9
      syscall
      cmpq $-1, %rax
      je .die

      # zero soul
      movq $SOUL_ADDR, %rdi
      xorq %rax, %rax
      movq $12, %rcx
      rep stosq

      movq $SOUL_ADDR, %rbx
      movq $EXPECTED_TSC, S_EXPECTED(%rbx)

      # initial TSC
      rdtscp
      shlq $32, %rdx
      orq  %rdx, %rax
      movq %rax, S_LAST_TSC(%rbx)
      movq %rax, S_CUR_TSC(%rbx)
      call crc_store

      xorq %r12, %r12                 # tick counter

  # ══════════════════════════════════════════════════════════════════
  # Heartbeat
  # ══════════════════════════════════════════════════════════════════
  .tick:
      # ── Sense ─────────────────────────────────────────────────────
      incq %r12
      movq $SOUL_ADDR, %rbx
      movl %r12d, S_TICK(%rbx)

      rdtscp
      shlq $32, %rdx
      orq  %rdx, %rax
      movq %rax, S_CUR_TSC(%rbx)

      movq S_LAST_TSC(%rbx), %rcx
      subq %rcx, %rax
      movq %rax, S_ELAPSED(%rbx)

      # ── Compare ───────────────────────────────────────────────────
      # |elapsed - expected| > expected/10 → hw_stress = 1, else 0
      movb $0, S_HW_STRESS(%rbx)
      movq S_ELAPSED(%rbx), %rax
      movq S_EXPECTED(%rbx), %rcx
      subq %rcx, %rax                  # deviation (signed)
      movq %rax, %rdx
      negq %rax
      cmovs %rdx, %rax                 # rax = |deviation|

      pushq %rax
      movq S_EXPECTED(%rbx), %rax
      xorq %rdx, %rdx
      movq $10, %rcx
      divq %rcx                        # rax = expected/10
      movq %rax, %rcx                  # threshold
      popq  %rax

      cmpq %rcx, %rax
      jbe  .cmp_ok
      movb $1, S_HW_STRESS(%rbx)
  .cmp_ok:

      # ── React ─────────────────────────────────────────────────────
      movq S_CUR_TSC(%rbx), %rax
      movq %rax, S_LAST_TSC(%rbx)

      # ── Record ────────────────────────────────────────────────────
      call crc_store

      # ── Verify ────────────────────────────────────────────────────
      call crc_check
      testq %rax, %rax
      jnz  .vfy_ok

      # checksum bad → wipe & restart
      movq $SOUL_ADDR, %rdi
      xorq %rax, %rax
      movq $12, %rcx
      rep stosq
      movq $SOUL_ADDR, %rbx
      movq $EXPECTED_TSC, S_EXPECTED(%rbx)
      xorq %r12, %r12
      rdtscp
      shlq $32, %rdx
      orq  %rdx, %rax
      movq %rax, S_LAST_TSC(%rbx)
      movq %rax, S_CUR_TSC(%rbx)
      call crc_store
      jmp  .tick
  .vfy_ok:

      # ── Output ────────────────────────────────────────────────────
      movq $SYS_WRITE, %rax
      movq $1, %rdi
      leaq dot(%rip), %rsi
      movq $1, %rdx
      syscall

      # every 100 ticks: print "tick N"
      movq %r12, %rax
      xorq %rdx, %rdx
      movq $100, %rcx
      divq %rcx
      testq %rdx, %rdx
      jnz  .no_tick

      # format "tick NNN\n"
      leaq tbuf(%rip), %rdi
      leaq prefix(%rip), %rsi
      movq $5, %rcx
  .rep: movb (%rsi),%al; movb %al,(%rdi); incq %rsi; incq %rdi; decq %rcx; jnz
  .rep

      movq %r12, %rax
      xorq %r8, %r8
  .itoa: xorq %rdx, %rdx; movq $10,%rcx; divq %rcx; pushq %rdx; incq %r8; testq
  %rax,%rax; jnz .itoa
  .out:  popq  %rax; addb $'0',%al; movb %al,(%rdi); incq %rdi; decq %r8; jnz
  .out
      movb $10, (%rdi)
      leaq tbuf(%rip), %rsi
      movq %rdi, %rax
      subq %rsi, %rax
      movq %rax, %rdx
      movq $SYS_WRITE, %rax
      movq $1, %rdi
      syscall
  .no_tick:

      # ── Wait (busy-wait with pause) ───────────────────────────────
      movq $SOUL_ADDR, %rbx
  .wait: pause
         rdtscp
         shlq $32, %rdx
         orq  %rdx, %rax
         subq S_LAST_TSC(%rbx), %rax
         cmpq S_EXPECTED(%rbx), %rax
         jl   .wait

      jmp  .tick

  # ══════════════════════════════════════════════════════════════════
  # CRC32 helpers — software, polynomial 0xEDB88320
  # ══════════════════════════════════════════════════════════════════

  # crc_store: compute CRC32 over soul (zeroing crc field), store it
  crc_store:
      pushq %rbx
      movq  $SOUL_ADDR, %rbx
      movl  $0, S_CRC(%rbx)
      movq  %rbx, %rsi
      movq  $96, %rcx
      xorl  %eax, %eax
      notl  %eax
  .cs_b: movzbl (%rsi), %edi; xorl %edi, %eax
         movq $8, %rdx
  .cs_bit: shrl $1, %eax; jnc .cs_nb; xorl $0xEDB88320, %eax
  .cs_nb:  decq %rdx; jnz .cs_bit
         incq %rsi; decq %rcx; jnz .cs_b
      notl  %eax
      movl  %eax, S_CRC(%rbx)
      popq  %rbx
      retq

  # crc_check: verify CRC32. Return 1=ok 0=bad in %rax
  crc_check:
      pushq %rbx
      movq  $SOUL_ADDR, %rbx
      movl  S_CRC(%rbx), %r8d
      movl  $0, S_CRC(%rbx)
      movq  %rbx, %rsi
      movq  $96, %rcx
      xorl  %eax, %eax
      notl  %eax
  .cc_b: movzbl (%rsi), %edi; xorl %edi, %eax
         movq $8, %rdx
  .cc_bit: shrl $1, %eax; jnc .cc_nb; xorl $0xEDB88320, %eax
  .cc_nb:  decq %rdx; jnz .cc_bit
         incq %rsi; decq %rcx; jnz .cc_b
      notl  %eax
      movl  %r8d, S_CRC(%rbx)
      cmpl  %r8d, %eax
      sete  %al
      movzbq %al, %rax
      popq  %rbx
      retq

  .die:
      movq $SYS_EXIT, %rax
      movq $1, %rdi
      syscall

  245 行，纯 AT&T 语法，无 libc 依赖，编译 as -o tork_core.o tork_core.asm && ld
   -o tork_core tork_core.o。
